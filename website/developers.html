Code Examples

Python SDK

import devguardian_ai

# Initialize client
client = devguardian_ai.Client(api_key="your_api_key")

# Scan code
result = client.scan_code(
    code="SELECT * FROM users WHERE id = " + user_input,
    language="php"
)

# Get vulnerabilities
for vuln in result.vulnerabilities:
    print(f"Found {vuln.type} at line {vuln.line}")

# Generate fixes
if result.vulnerabilities:
    fixes = client.generate_fixes(result.vulnerabilities)
    for fix in fixes:
        print(f"Fix: {fix.description}")
        print(f"Code: {fix.fixed_code}")

JavaScript SDK

const DevGuardianAI = require('devguardian-ai');

// Initialize client
const client = new DevGuardianAI({
    apiKey: 'your_api_key'
});

// Scan code
const result = await client.scanCode({
    code: 'SELECT * FROM users WHERE id = ' + userInput,
    language: 'php'
});

// Process results
result.vulnerabilities.forEach(vuln => {
    console.log(`Found ${vuln.type} at line ${vuln.line}`);
});

// Generate fixes
if (result.vulnerabilities.length > 0) {
    const fixes = await client.generateFixes(result.vulnerabilities);
    fixes.forEach(fix => {
        console.log(`Fix: ${fix.description}`);
        console.log(`Code: ${fix.fixedCode}`);
    });
}
